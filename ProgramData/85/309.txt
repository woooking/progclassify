					printf("no\n");

				}
			}
			else
			{
				if((num[i][k]>='A' && num[i][k]<='Z') || (num[i][k]>='a' && num[i][k]<='z') || (num[i][k]>='0' && num[i][k]<='9') || num[i][k]=='_') 
				{
					tmp++;
				}
				else
				{
					printf("no\n");
					break;
				}
			}
			k++;
		}
		if(tmp==len) printf("yes\n");
	}
}
